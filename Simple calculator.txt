    <script>
        // دریافت المان‌های مورد نیاز
        const currentOperandElement = document.getElementById('current-operand');
        const previousOperandElement = document.getElementById('previous-operand');
        const historyListElement = document.getElementById('history-list');
        const clearHistoryButton = document.getElementById('clear-history');

        // متغیرهای ماشین حساب
        let currentOperand = '0';
        let previousOperand = '';
        let operation = null;
        let resetCurrentOperand = false;

        // تاریخچه عملیات
        let calculationHistory = [];

        // به روزرسانی نمایشگر
        function updateDisplay() {
            currentOperandElement.textContent = currentOperand;
            
            if (operation) {
                previousOperandElement.textContent = `${previousOperand} ${getOperationSymbol(operation)}`;
            } else {
                previousOperandElement.textContent = previousOperand;
            }
        }

        // دریافت نماد عملگر
        function getOperationSymbol(op) {
            switch(op) {
                case 'add': return '+';
                case 'subtract': return '−';
                case 'multiply': return '×';
                case 'divide': return '÷';
                default: return '';
            }
        }

        // اضافه کردن عدد
        function appendNumber(number) {
            if (currentOperand === '0' || resetCurrentOperand) {
                currentOperand = number;
                resetCurrentOperand = false;
            } else {
                if (number === '.' && currentOperand.includes('.')) return;
                currentOperand += number;
            }
            updateDisplay();
        }

        // انتخاب عملگر
        function chooseOperation(op) {
            if (currentOperand === '') return;
            
            if (previousOperand !== '') {
                calculate();
            }
            
            operation = op;
            previousOperand = currentOperand;
            resetCurrentOperand = true;
            updateDisplay();
        }

        // محاسبه
        function calculate() {
            let result;
            const prev = parseFloat(previousOperand);
            const current = parseFloat(currentOperand);
            
            if (isNaN(prev) || isNaN(current)) return;
            
            switch(operation) {
                case 'add':
                    result = prev + current;
                    break;
                case 'subtract':
                    result = prev - current;
                    break;
                case 'multiply':
                    result = prev * current;
                    break;
                case 'divide':
                    if (current === 0) {
                        alert("تقسیم بر صفر امکان پذیر نیست!");
                        clear();
                        return;
                    }
                    result = prev / current;
                    break;
                default:
                    return;
            }
            
            // ذخیره در تاریخچه
            const calculation = {
                expression: `${previousOperand} ${getOperationSymbol(operation)} ${currentOperand}`,
                result: result.toString()
            };
            
            calculationHistory.unshift(calculation);
            if (calculationHistory.length > 8) {
                calculationHistory.pop();
            }
            
            updateHistoryDisplay();
            
            currentOperand = result.toString();
            operation = null;
            previousOperand = '';
            resetCurrentOperand = true;
            updateDisplay();
        }

        // پاک کردن
        function clear() {
            currentOperand = '0';
            previousOperand = '';
            operation = null;
            updateDisplay();
        }

        // حذف آخرین رقم
        function backspace() {
            if (currentOperand.length > 1) {
                currentOperand = currentOperand.slice(0, -1);
            } else {
                currentOperand = '0';
            }
            updateDisplay();
        }

        // درصد
        function percentage() {
            if (currentOperand === '0') return;
            const value = parseFloat(currentOperand);
            currentOperand = (value / 100).toString();
            updateDisplay();
        }

        // به روزرسانی تاریخچه
        function updateHistoryDisplay() {
            historyListElement.innerHTML = '';
            
            if (calculationHistory.length === 0) {
                historyListElement.innerHTML = '<div class="history-item">هیچ عملیاتی انجام نشده است</div>';
                return;
            }
            
            calculationHistory.forEach(item => {
                const historyItem = document.createElement('div');
                historyItem.className = 'history-item';
                historyItem.textContent = `${item.expression} = ${item.result}`;
                historyListElement.appendChild(historyItem);
            });
        }

        // پاک کردن تاریخچه
        function clearHistory() {
            calculationHistory = [];
            updateHistoryDisplay();
        }

        // کنترل با صفحه کلید
        function handleKeyboardInput(e) {
            const key = e.key;
            
            if ((key >= '0' && key <= '9') || key === '.') {
                appendNumber(key);
            } else if (key === '+') {
                chooseOperation('add');
            } else if (key === '-') {
                chooseOperation('subtract');
            } else if (key === '*') {
                chooseOperation('multiply');
            } else if (key === '/') {
                e.preventDefault();
                chooseOperation('divide');
            } else if (key === 'Enter' || key === '=') {
                e.preventDefault();
                calculate();
            } else if (key === 'Escape' || key === 'Delete') {
                clear();
            } else if (key === 'Backspace') {
                backspace();
            } else if (key === '%') {
                percentage();
            }
        }

        // رویداد کلیک برای دکمه‌ها
        document.querySelectorAll('.btn').forEach(button => {
            button.addEventListener('click', () => {
                if (button.classList.contains('number')) {
                    const number = button.getAttribute('data-number');
                    appendNumber(number);
                }
                
                if (button.classList.contains('operator')) {
                    const action = button.getAttribute('data-action');
                    
                    if (action === 'clear') {
                        clear();
                    } else if (action === 'backspace') {
                        backspace();
                    } else if (action === 'percentage') {
                        percentage();
                    } else {
                        chooseOperation(action);
                    }
                }
                
                if (button.classList.contains('equals')) {
                    calculate();
                }
            });
        });

        // رویداد برای پاک کردن تاریخچه
        clearHistoryButton.addEventListener('click', clearHistory);

        // رویداد صفحه کلید
        document.addEventListener('keydown', handleKeyboardInput);

        // مقداردهی اولیه
        updateDisplay();
        updateHistoryDisplay();
    </script>
